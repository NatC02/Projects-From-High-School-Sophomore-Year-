//if else example using strings

public class IfElseString
{
	public static void main(String args[])
	{
		String s = "one";
		if(s.equals("one"))
		{
		   System.out.println(s + " is one!");
		}
		else
		{
		   System.out.println(s + " is not one!");
		}
	}
}
